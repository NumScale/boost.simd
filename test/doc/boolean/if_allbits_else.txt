---- simd
 <- pf =                          (3, -0, -3, 0)
 <- qf =                          (4, -3, -3, 0)
 -> bs::if_allbits_else(pf, qf) = (-nan, -3, -nan, 0)
---- scalar
 xf =                             3
 yf =                             4
 -> bs::if_allbits_else(xf, yf) = -nan
